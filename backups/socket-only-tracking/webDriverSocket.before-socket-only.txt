import {
  io,
  type Socket,
} from "socket.io-client";

import type { WebBusLocation } from "../types/driverTracking";

const SERVER_URL =
  "https://pop-bus-server.onrender.com";

export type WebDriverSocketStatus =
  | "connecting"
  | "connected"
  | "disconnected"
  | "error";

type StatusListener = (
  status: WebDriverSocketStatus
) => void;

let socket: Socket | null = null;
let currentStatusListener:
  | StatusListener
  | null = null;

function notifyStatus(
  status: WebDriverSocketStatus
) {
  currentStatusListener?.(status);
}

function createSocket() {
  const nextSocket = io(SERVER_URL, {
    autoConnect: false,
    transports: ["websocket"],
    reconnection: true,
    reconnectionAttempts: Infinity,
    reconnectionDelay: 1000,
    reconnectionDelayMax: 5000,
    timeout: 15000,
  });

  nextSocket.on("connect", () => {
    notifyStatus("connected");
  });

  nextSocket.on("disconnect", () => {
    notifyStatus("disconnected");
  });

  nextSocket.on("connect_error", () => {
    notifyStatus("error");
  });

  nextSocket.on("server:error", () => {
    notifyStatus("error");
  });

  return nextSocket;
}

export function connectWebDriverSocket(
  onStatusChange?: StatusListener
) {
  currentStatusListener =
    onStatusChange ?? null;

  if (!socket) {
    socket = createSocket();
  }

  if (socket.connected) {
    notifyStatus("connected");
    return socket;
  }

  notifyStatus("connecting");
  socket.connect();

  return socket;
}

export function isWebDriverSocketConnected() {
  return Boolean(socket?.connected);
}

export function sendWebDriverLocation(
  location: WebBusLocation,
  reliable = false
) {
  if (!socket?.connected) {
    return false;
  }

  if (reliable) {
    socket.emit(
      "driver:location-update",
      location
    );
  } else {
    socket.volatile.emit(
      "driver:location-update",
      location
    );
  }

  return true;
}

export function stopWebDriverSocketSharing(
  busId: string
) {
  return new Promise<boolean>((resolve) => {
    if (!socket?.connected) {
      resolve(false);
      return;
    }

    const timeout = window.setTimeout(() => {
      resolve(false);
    }, 3500);

    socket.emit(
      "driver:stop-sharing",
      { busId },
      (response?: { ok?: boolean }) => {
        window.clearTimeout(timeout);
        resolve(response?.ok === true);
      }
    );
  });
}

export function disconnectWebDriverSocket() {
  currentStatusListener = null;

  if (!socket) {
    return;
  }

  socket.disconnect();
  socket = null;
}
